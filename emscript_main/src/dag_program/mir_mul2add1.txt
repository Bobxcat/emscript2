struct TwoInts {

}

fn add_inner(arg0) -> i32 {
    let arg0: &TwoInts;
    let v1: i32;
    bb0: {
        v1 = add (*arg0).0, (*arg0).1;
        return v1
    }
}

fn i32_mul2add11(arg0) -> i32 {
    let arg0: i32;
    let v0: TwoInts;
    let v1: i32;
    let v2: &TwoInts;
    bb0: {
        v0.0 = add arg0, 1;
        v0.1 = arg0;
        v2 = &v0;
        call add_inner, v1, bb1, v2
